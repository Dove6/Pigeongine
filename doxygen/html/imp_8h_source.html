73 FLAC__StreamDecoderWriteStatus sound::FLAC_write(const FLAC__StreamDecoder *decoder,
74  const FLAC__Frame *frame,
75  const FLAC__int32 * const buffer[],
76  void *client_data)
77 {
78  sound *snd = (sound *)client_data;
79  (void)decoder;
80 
81  if(snd->metadata.totalSamples == 0) {
82  fprintf(stderr, "FLAC ERROR: no total_samples count in STREAMINFO\n");
83  return FLAC__STREAM_DECODER_WRITE_STATUS_ABORT;
84  }
85 
86  for (int channel = 0; channel < snd->metadata.channels; channel++) {
87  if(buffer[channel] == NULL) {
88  fprintf(stderr, "FLAC ERROR: buffer [%d] of %d is NULL\n", channel, snd->metadata.channels );
89  return FLAC__STREAM_DECODER_WRITE_STATUS_ABORT;
90  }
91  }
92 
93  switch (snd->metadata.bps) {
94  case 4:
95  case 8: {
96  FLAC_write_int8(frame, buffer, snd);
97  break;
98  } /* 8 bit unsigned integer output */
99  case 16: {
100  FLAC_write_int16(frame, buffer, snd);
101  break;
102  } /* 16 bit signed integer output */
103  case 24: {
104  FLAC_write_int32(frame, buffer, snd);
105  break;
106  } /* 24 bit signed integer output */
107  default: {
108  FLAC_write_int32(frame, buffer, snd);
109  }
110  }
111 
112  return FLAC__STREAM_DECODER_WRITE_STATUS_CONTINUE;
113 }

192 {
193  int8_t *out = (int8_t *)outputBuffer;

196  if (snd->pos < snd->metadata.totalSamples) {
197  if (snd->metadata.channels < snd->deviceInfo->maxOutputChannels) {
198  for(int channel = 0, hw_channel = 0; hw_channel < snd->deviceInfo->maxOutputChannels; hw_channel++) {
199  data = (int8_t *)snd->data[channel];
200  *out++ = data[snd->pos] * snd->metadata.volume;
201  if (channel >= snd->metadata.channels) channel = 0;
202  }
203  snd->pos++;
204  }
205  else {
206  for(int channel = 0; channel < snd->metadata.channels; channel++) {
207  data = (int8_t *)snd->data[channel];
208  *out++ = data[snd->pos] * snd->metadata.volume;
209  }
210  snd->pos++;
211  }
212  }
213  else {
214  if (snd->metadata.channels < snd->deviceInfo->maxOutputChannels) {
215  for(int channel = 0; channel < snd->deviceInfo->maxOutputChannels; channel++)
216  *out++ = 0;
217  snd->pos = 0;
218  }
219  else {
220  for(int channel = 0; channel < snd->metadata.channels; channel++)
221  *out++ = 0;
222  snd->pos = 0;
223  }
224  #ifdef DEBUG

227  }

359 //Funkcja odpowiadająca za wczytanie i dekompresję danych dźwiękowych formatu FLAC.
360 int sound::FLAC_decode(sound *snd)
361 {
362  if (snd->filename[0] != '\0') {
363  FLAC__bool ok = true;
364  FLAC__StreamDecoder *decoder = 0;
365  FLAC__StreamDecoderInitStatus initStatus;
366 
367  if ((decoder = FLAC__stream_decoder_new()) == NULL) {
368  fprintf(stderr, "ERROR: allocating decoder\n");
369  return -1;
370  }
371 
372  (void)FLAC__stream_decoder_set_md5_checking(decoder, true);
373 
374  initStatus = FLAC__stream_decoder_init_file(decoder, snd->filename, FLAC_write, FLAC_meta, FLAC_error, snd /*user*/);
375  if(initStatus != FLAC__STREAM_DECODER_INIT_STATUS_OK) {
376  fprintf(stderr, "ERROR: initializing decoder: %s\n", FLAC__StreamDecoderInitStatusString[initStatus]);
377  ok = false;
378  }
379 
380  if(ok) {
381  ok = FLAC__stream_decoder_process_until_end_of_stream(decoder);
382  fprintf(stderr, "decoding: %s\n", ok? "succeeded" : "FAILED");
383  fprintf(stderr, " state: %s\n", FLAC__StreamDecoderStateString[FLAC__stream_decoder_get_state(decoder)]);
384  }
385 
386  FLAC__stream_decoder_delete(decoder);
387 
388  #ifdef DEBUG
389  printf("End of decoding file\n\n");
390  #endif // DEBUG
391 
392  return ok? 0 : -1;
393  }
394  else return -1;
395 }
